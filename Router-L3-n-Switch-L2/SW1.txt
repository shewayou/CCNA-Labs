config t

!
!
! SW1: L2 Switch 2960-24TT
! SW1 interact w R1 ISR4331 L3 port
! SW1 interact w SW2 SW3 SW4 L2 switch 2960-24TT
! "inter-vlan routing, option 2, Router-on-a-stick" method

no ip domain-lookup

hostname SW1
 
username admin secret ccna123
enable secret ccna124

ip domain-name lab.local

ip ssh version 2

crypto key generate rsa  general-keys modulus 2048

! via rollover cable
line con 0
 exec-timeout 5 0
 privilege level 15
 logging synchronous
 exit

! via Virtual Teletype
line vty 0 4
 transport input ssh
 login local
 logging synchronous
 exit


vlan 10
name Eng
vlan 20
name Sales
vlan 30
name Admin
vlan 99
name Management
vlan 199
name Native
exit



! 1. Create SVI for SW1 Management IP
interface vlan 99
 description SW1 Management SVI
 ip address 192.168.99.11 255.255.255.0
 ipv6 address 2001:db8:0:99::11/64
 ipv6 address fe80::21 link-local
 no shutdown
 exit
 
! 2. Configure Default Gateway (points to R1's subinterface IP)
ip default-gateway 192.168.99.1


! SW1 -- R1
! gig0/1 is L2 ports
interface gigabitethernet0/1
 description SW1 -- R1 
 switchport mode trunk
 switchport trunk native vlan 199
 switchport trunk allowed vlan 10,20,30,99
 no shutdown
 exit

! SW1 -- SW2
! gig0/2 to SW2 L2 Switch 2960-24TT
interface gigabitethernet0/2
 description SW1 -- SW2 
 ! switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 199
 switchport trunk allowed vlan 10,20,99
 no shutdown
 exit
 
! SW1 -- SW3
! fa0/1 to SW3 L2 Switch 2960-24TT
interface fastethernet0/1
 description SW1 -- SW3 
 ! switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 199
 switchport trunk allowed vlan 10,30,99
 no shutdown
 exit
 
! SW1 -- SW4
! fa0/2 to SW4 L2 Switch 2960-24TT
interface fastethernet0/2
 description SW1 -- SW4 
 ! switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 199
 switchport trunk allowed vlan 20,30,99
 no shutdown
 exit
 
exit
