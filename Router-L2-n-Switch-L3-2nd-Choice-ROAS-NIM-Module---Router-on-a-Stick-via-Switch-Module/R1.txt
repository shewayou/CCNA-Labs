vlan database
vlan 999 name Transit_To_SW1
exit





config t

!
!
! R1: ISR4331 with 3 L3 ports n 4 NIM-E2-4 L2 ports
! R1 use built-in L3 port to connect with ISP.
! R1 use NIM-E2-4 L2 port to connect with SW1 L3 Switch 3650-24PS.
! "2nd Choice: ROAS-NIM-Module         Router-on-a-Stick-via-Switch-Module" method           or Case 4 2nd Choice implementation

no ip domain-lookup

hostname R1

! via rollover cable
line con 0
 exec-timeout 5 0
 privilege level 15
 logging synchronous
 exit

! Auxiliary port
! via out-of-band management via a dial-up modem
! traditionally, not in use for now
line aux 0
 logging synchronous
 exit
 
! via Virtual Teletype
line vty 0 4
 transport input ssh
 login local
 logging synchronous
 exit
 
 
username admin secret ccna123

ip domain-name lab.local

ip ssh version 2

crypto key generate rsa  general-keys modulus 2048

ipv6 unicast-routing


! 4. Static Routes pointing to SW1
ip route 192.168.0.0 255.255.0.0 10.0.0.2
ipv6 route 2001:db8:0::/48 2001:db8:0:0::2



 
! 2. Create the SVI to hold the IP address
interface vlan 999
 ip address 10.0.0.1 255.255.255.252
 ipv6 address 2001:db8:0:0::1/64
 ipv6 address fe80::1         link-local
 no shutdown
 exit 


! R1 -- ISP
! gig0/0/0 is L3 ports
interface gigabitethernet0/0/0
 description R1 -- ISP 
 ip   address 203.0.113.2   255.255.255.252
 ipv6 address 2001:db8:0:113::2/64
 ipv6 address fe80::1           link-local
 no shutdown
 exit
 

! The industry best practice for connecting a Layer 2 access port directly to a Layer 3 routed port 
! is to simply disable CDP on that specific link so it stops reporting the false positive. 
 
! R1 -- SW1
! gig0/1/0 is NIM-E2-4 L2 ports
interface gigabitethernet0/1/0
 description R1 -- SW1 (Transit Link)
 no cdp enable
 switchport mode access
 switchport access vlan 999
 no shutdown
 exit
 
interface loopback0
 description R1 loopback0
 ip   address 192.168.1.1         255.255.255.255
 ipv6 address 2001:db8:0:1::1/128 
 ipv6 address fe80::1         link-local
 exit
 
exit
