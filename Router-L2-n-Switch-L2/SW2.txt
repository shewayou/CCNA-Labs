config t

no ip domain-lookup

hostname SW2
 
username admin secret ccna123
enable secret ccna124

ip domain-name lab.local

ip ssh version 2

crypto key generate rsa  general-keys modulus 2048

! via rollover cable
line con 0
 exec-timeout 5 0
 privilege level 15
 logging synchronous
 exit

! via Virtual Teletype
line vty 0 4
 transport input ssh
 login local
 logging synchronous
 exit
 
! Create the required VLANs
vlan 10
 name Eng
vlan 20 
 name Sales
vlan 99
 name Management
vlan 199
 name Native
exit


interface vlan 99
 ip   address 192.168.99.2 255.255.255.0
 ipv6 address 2001:db8:0:99::2/64
 ipv6 address fe80::3         link-local
 no shutdown
 exit
 
ip default-gateway 192.168.99.101


! Trunk link to SW1
interface gigabitethernet0/2
 switchport mode trunk
 switchport trunk native vlan 199
 switchport trunk allowed vlan 10,20,99
 no shutdown
 exit

! Access Port for Eng1 (VLAN 10)
interface fastethernet0/11
 switchport mode access
 switchport access vlan 10
 spanning-tree portfast
 spanning-tree bpduguard
 no shutdown
 exit

! Access Port for Sales1 (VLAN 20)
interface fastethernet0/12
 switchport mode access
 switchport access vlan 20
 spanning-tree portfast
 spanning-tree bpduguard
 no shutdown
 exit
 
exit
